package com.example.finalapp;

import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.BaseAdapter;
import android.widget.TextView;

import com.blogspot.atifsoftwares.circularimageview.CircularImageView;
import com.bumptech.glide.Glide;

import java.util.ArrayList;

public class Distribution_WaitingList_Adapter extends BaseAdapter {

    Context contex;
    ArrayList<User> usersList;

    public Distribution_WaitingList_Adapter(Context c, ArrayList<User> usersList)
    {
        this.contex = c;
        this.usersList = usersList;
    }
    @Override
    public int getCount() {
        return usersList.size();
    }

    @Override
    public Object getItem(int position) {
        return null;
    }

    @Override
    public long getItemId(int position) {
        return 0;
    }

    @Override
    public View getView(int position, View convertView, ViewGroup parent) {
        LayoutInflater layoutInflater = (LayoutInflater) this.contex.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
        View show_Discarabution = layoutInflater.inflate(R.layout.distribution_waitinglist_row, parent, false);
        TextView fullname= (TextView)show_Discarabution.findViewById(R.id.email_userwaitingListlist);
        TextView username= (TextView)show_Discarabution.findViewById(R.id.name_userwaitingListlist);
        CircularImageView img= (CircularImageView) show_Discarabution.findViewById(R.id.img_userwaitingList);

        if(!usersList.get(position).getPic().equals("profile.png")){
            Glide.with(contex).load(usersList.get(position).getPic()).into(img);
        }


        fullname.setText(usersList.get(position).getEmail());
        username.setText(usersList.get(position).getName());


        return show_Discarabution;
    }
}
